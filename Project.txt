CREATE TABLE IF NOT EXISTS public.books
(
    isbn13 character varying(13) COLLATE pg_catalog."default" NOT NULL,
    isbn10 character varying(10) COLLATE pg_catalog."default",
    title character varying(250) COLLATE pg_catalog."default",
    subtitle character varying(250) COLLATE pg_catalog."default",
    authors character varying(250) COLLATE pg_catalog."default",
    categories character varying(250) COLLATE pg_catalog."default",
    thumbnail character varying(1000) COLLATE pg_catalog."default",
    description text COLLATE pg_catalog."default",
    published_year integer,
    average_rating real,
    num_pages integer,
    ratings_count integer,
    CONSTRAINT books_pkey PRIMARY KEY (isbn13)
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.books
    OWNER to postgres;
